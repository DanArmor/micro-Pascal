17std::unique_ptr<IAST> ASTFactory::createAST(Token token){
18 switch (token.getAdvType()){
19 case IToken::TYPE_SPEC: return std::unique_ptr<IAST>(new TypeSpecAST(token));
20 }
21 switch (token.getType()){
22 case IToken::STRING_CONST: return std::unique_ptr<IAST>(new StringAST(token));
23 case IToken::INTEGER_CONST: return std::unique_ptr<IAST>(new NumberAST(token));
24 case IToken::REAL_CONST: return std::unique_ptr<IAST>(new NumberAST(token));
25 case IToken::ID: return std::unique_ptr<IAST>(new VarAST(token));
26 case IToken::EMPTY: {
27 token.setStr("$");
28 return std::unique_ptr<IAST>(new NoOpAST(token));
29 }
30 }
31 throw std::runtime_error(fmt::format("Попытка создания AST узла из неверного токена и набора аргументов! {}", token.getInfo()));
32}

85 return std::move(compound);
