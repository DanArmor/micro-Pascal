10std::unique_ptr<AST> ASTFactory::createAST(Token token){
11 switch (token.getAdvType()){
12 case IToken::TYPE_SPEC: return std::unique_ptr<AST>(new TypeSpecAST(token));
13 }
14 switch (token.getType()){
15 case IToken::STRING_CONST: return std::unique_ptr<AST>(new StringAST(token));
16 case IToken::INTEGER_CONST: return std::unique_ptr<AST>(new NumberAST(token));
17 case IToken::REAL_CONST: return std::unique_ptr<AST>(new NumberAST(token));
18 case IToken::ID: return std::unique_ptr<AST>(new VarAST(token));
19 case IToken::EMPTY: {
20 token.setStr("$");
21 return std::unique_ptr<AST>(new NoOpAST(token));
22 }
23 }
24 return std::unique_ptr<AST>(new NoOpAST(token));
25}
