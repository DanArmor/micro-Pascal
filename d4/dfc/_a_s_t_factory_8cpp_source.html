9std::unique_ptr<IAST> ASTFactory::createAST(Token token) {
10 switch (token.getAdvType()) {
11 case IToken::TYPE_SPEC:
12 return std::unique_ptr<IAST>(new TypeSpecAST(token));
13 }
14 switch (token.getType()) {
15 case IToken::STRING_CONST:
16 return std::unique_ptr<IAST>(new StringAST(token));
17 case IToken::INTEGER_CONST:
18 return std::unique_ptr<IAST>(new NumberAST(token));
19 case IToken::REAL_CONST:
20 return std::unique_ptr<IAST>(new NumberAST(token));
21 case IToken::ID:
22 return std::unique_ptr<IAST>(new VarAST(token));
23 case IToken::EMPTY: {
24 token.setStr("$");
25 return std::unique_ptr<IAST>(new NoOpAST(token));
26 }
27 }
28 throw std::runtime_error(fmt::format(
29 "Попытка создания AST узла из неверного токена и набора аргументов! {}",
30 token.getInfo()));
31}

58 switch (token.getAdvType()) {
59 case IToken::OPERATOR:

61 new BinOpAST(token, std::move(first), std::move(second)));
62 case IToken::VARDECL: {

66 }
67 case IToken::SELECT: {

71 }
72 case IToken::CONSTDECL: {
73 token.setStr("CONSTDECL");
74 return std::unique_ptr<IAST>(
75 new ConstAST(token, std::move(first), std::move(second)));
76 }
77 }
78 throw std::runtime_error(fmt::format(

105std::unique_ptr<IAST> ASTFactory::createAST(
106 Token token, std::vector<std::unique_ptr<IAST>> params) {
107 switch (token.getAdvType()) {
108 case IToken::AdvType::COMPOUND: {
109 token.setStr("COMPOUND");
110 std::unique_ptr<CompoundAST> compound(new CompoundAST(token));
111 for (auto &ptr : params) compound->addChild(std::move(ptr));
112 return std::move(compound);
113 }
114 case IToken::AdvType::FUNCTION_NAME:
115 return std::unique_ptr<IAST>(new CallAST(token, std::move(params)));
116 }
117 throw std::runtime_error(fmt::format(
118 "Попытка создания AST узла из неверного токена и набора аргументов! {}",
119 token.getInfo()));
120}

175std::unique_ptr<IAST> ASTFactory::createCopyOfType(IAST *typePtr) {
176 if (typePtr->token.getAdvType() == IToken::TYPE_SPEC ||
177 typePtr->token.getType() == IToken::ARRAY) {
178 if (typePtr->token.getType() == IToken::ARRAY) {
179 ArrSpecAST *rawPtr = static_cast<ArrSpecAST *>(typePtr);
180 std::unique_ptr<IAST> arrPtr =
181 createAST(rawPtr->token, rawPtr->lHandTok, rawPtr->rHandTok,
182 createCopyOfType(rawPtr->subType.get()));
183 return arrPtr;
184 } else {
185 return createAST(typePtr->token);
186 }
187 } else {
188 throw std::logic_error(
189 "Применение createCopyOfType к узлу, не являющемуся типом! " +
190 typePtr->token.getInfo());
191 }
