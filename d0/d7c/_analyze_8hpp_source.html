27void highlight(std::string text, List<Token> tokens){
28 std::map<IToken::AdvType, std::string> colors = {
29 {IToken::AdvType::SOME_CONST, "\033[31m"},
30 {IToken::AdvType::FUNCTION_NAME, "\033[32m"},
31 {IToken::AdvType::VAR_NAME, "\033[33m"},
32 {IToken::AdvType::TYPE_SPEC, "\033[1;34m"},
33 {IToken::AdvType::KEYWORD, "\033[35m"},
34 {IToken::AdvType::OPERATOR, "\033[36m"},
35 {IToken::AdvType::PROGRAM_NAME, "\033[1;31m"},
36 {IToken::AdvType::UNKNOWN, "\033[4;37m"},
37 {IToken::AdvType::NOTPROCESS, "\033[37m"},
38 };
39 int tokenI = 0;
40 int i = 0;
41 while(i < text.size()){
42 if(tokenI < tokens.size()-1 && i == tokens[tokenI].pos()){
43 std::cout << "\033[0m" << colors[tokens[tokenI].getAdvType()];
44 std::size_t len = tokens[tokenI].len();
45 while(len != 0){
46 std::cout << text[i];
47 i++;
48 len--;
49 }
50 tokenI++;
51 } else{
52 std::cout << "\033[0m\033[30;1m";
53 if(tokenI < tokens.size() - 1){
54 while(i != tokens[tokenI].pos()){
55 std::cout << text[i];
56 i++;
57 }
58 }
59 else{
60 while(i < text.size()){
61 std::cout << text[i];
62 i++;
63 }
64 }
65 }
66 }
67 std::cout << "\033[0m\n";
68}
