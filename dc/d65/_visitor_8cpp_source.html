397void SemanticVisitor::visit(BinOpAST &node){
398 std::string lHand = getValue(node.left.get());
399 std::string rHand = getValue(node.right.get());
400 if(!isIn(lHand, {std::string("integer"), std::string("real")})){
401 throw TypeException(node.token, fmt::format("Ожидался числовой тип, а получен {}. ", lHand));
402 }
403 if(!isIn(rHand, {std::string("integer"), std::string("real")})){
404 throw TypeException(node.token, fmt::format("Ожидался числовой тип, а получен {}. ", rHand));
405 }
406 switch(node.token.getType()){
407 case IToken::MOD:
408 if(lHand != "integer" || rHand != "integer"){
409 throw TypeException(node.token, fmt::format("MOD определен только над целыми операндами, а получены {} и {}. ", lHand, rHand));
410 }
411 case IToken::LESS:
412 case IToken::LESS_EQ:
413 case IToken::EQ:
414 case IToken::MORE:
415 case IToken::MORE_EQ:
416 case IToken::OR:
417 case IToken::AND:
418 case IToken::NOT:
419 Return("integer");
420 break;
421 default:
422 if(lHand == "real" || rHand == "real")
423 Return("real");
424 else
425 Return("integer");
426 }
427}

474void SemanticVisitor::visit(CallAST &node){

612bool SemanticVisitor::compareTypes(std::string A, std::string B, bool strict) {
613 if (strict) {
614 return A == B;
615 } else {
616 if (isIn(A, {std::string("integer"), std::string("real")}) && isIn(B, {std::string("integer"), std::string("real")})) {
617 return true;
