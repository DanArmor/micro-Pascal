130ASTptr SyntaxAnalyzer::syntaxSt(void){
131 /* syntaxSt ::= (syntaxCompoundSt|'<ID>'(syntaxAssignSt|syntaxCallSt)|syntaxWhileSt|syntaxIfSt|syntaxEmptySt) */
132 switch(getCurTok().getType()) {
133 case IToken::BEGIN :
134 return syntaxCompoundSt();
135 case IToken::ID :{
136 if(lookFoward().getType() == IToken::ASSIGN)
137 return syntaxAssignSt();
138 else if(lookFoward().getType() == IToken::LPAREN)
139 return syntaxCallSt();
140 else{
141 // Специально форсим, будто ждали присваивание
142 eat(IToken::ID);
143 eat(IToken::ASSIGN);
144 }
145 }
146 case IToken::WHILE :
147 return syntaxWhileSt();
148 case IToken::IF :
149 return syntaxIfSt();
150 default:
151 return syntaxEmptySt();
152 }
153}

268AST *SyntaxAnalyzer::syntaxFactor(){
269 /* syntaxFactor ::= ('<INTEGER_CONST>'|'<REAL_CONST>'|'<STRING_CONST>'
270 |'(' syntaxExpr ')'
271 |'+' syntaxFactor| '-' syntaxFactor
272 |'NOT' syntaxFactor
273 |syntaxCallSt| syntaxVariable) */
274 Token token = getCurTok();
275 if(token.getAdvType() == IToken::SOME_CONST){
276 eatAdv(IToken::SOME_CONST);
277 return ASTFactory::createAST(token);
278 }
279 switch(token.getType()) {
280 case IToken::LPAREN :{
281 eat(IToken::LPAREN);
282 AST *node = syntaxExpr();
283 eat(IToken::RPAREN);
284 return node;
285 }
286
287 case IToken::PLUS :
288 case IToken::MINUS :
289 case IToken::NOT :{
290 eatAdv(IToken::OPERATOR);
291 return ASTFactory::createAST(token, syntaxFactor());
292 }
293
294 case IToken::ID :{
295 if(lookFoward().getType() == IToken::LPAREN)
296 return syntaxCallSt();
297 else
298 return syntaxVariable();
299 }
300 default:
301 throw std::invalid_argument(fmt::format("В syntaxFactor попал лишний токен: {}", token.getInfo()));
302 }
303 return nullptr;
304}
