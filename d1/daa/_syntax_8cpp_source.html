18std::unique_ptr<IAST> SyntaxAnalyzer::syntaxProgram(void){
19 /* syntaxProgram ::= 'PROGRAM' 'ID' ';' syntaxBlock '.' */
20 eat(IToken::PROGRAM);
21 Token &nameTok = getCurTok();
22 nameTok.setAdvType(IToken::PROGRAM_NAME);
23
24 eat(IToken::ID);
25 eat(IToken::SEMI);
26 std::vector<std::unique_ptr<IAST>> functions;
27
28 while(isIn(getCurTok().getType(), {IToken::PROCEDURE, IToken::FUNCTION})){
29 functions.emplace_back(std::move(syntaxFuncDef()));
30 eat(IToken::SEMI);
31 }
32
33 std::unique_ptr<IAST> blockPTR = syntaxBlock();
34 eat(IToken::DOT);
35 return ASTFactory::createAST(nameTok, std::move(functions), std::move(blockPTR));
36}

38std::unique_ptr<IAST> SyntaxAnalyzer::syntaxFuncDef(void){
39 Token funcTok = getCurTok();
40 if(funcTok.getType() == IToken::FUNCTION)
41 eat(IToken::FUNCTION);
42 else
43 eat(IToken::PROCEDURE);
44
45 Token &name = getCurTok();
46
47 name.setAdvType(IToken::FUNCTION_NAME);
48 eat(IToken::ID);
49 eat(IToken::LPAREN);
50 std::vector<std::unique_ptr<IAST>> params;
51 while(getCurTok().getType() != IToken::RPAREN){
52 std::vector<std::unique_ptr<IAST>> sup = std::move(syntaxVarDecl());
53 for(auto &ptr : sup)
54 params.emplace_back(std::move(ptr));
55 if(getCurTok().getType() == IToken::COMMA)
56 eat(IToken::COMMA);
57 }
58 eat(IToken::RPAREN);
59
60 std::unique_ptr<IAST> returnType;
61 if(funcTok.getType() == IToken::FUNCTION){
62 eat(IToken::COLON);
63 returnType = syntaxTypeSpec();
64 } else{
65 Token typeTok = {"VOID", IToken::VOID, IToken::TYPE_SPEC};
66 returnType = ASTFactory::createAST(typeTok);
67 }
68 eat(IToken::SEMI);
69 std::unique_ptr<IAST> body = std::move(syntaxBlock());
70 return ASTFactory::createAST(name, std::move(params), std::move(returnType), std::move(body));
71}

202std::unique_ptr<IAST> SyntaxAnalyzer::syntaxSt(void){
203 /* syntaxSt ::= (syntaxCompoundSt|'<ID>'(syntaxAssignSt|syntaxCallSt)|syntaxWhileSt|syntaxIfSt|syntaxEmptySt) */
204 switch(getCurTok().getType()) {
205 case IToken::BEGIN :
206 return syntaxCompoundSt();
207 case IToken::ID :{
208 if(lookFoward().getType() == IToken::ASSIGN || lookFoward().getType() == IToken::LSQBRACKET)
209 return syntaxAssignSt();
210 else if(lookFoward().getType() == IToken::LPAREN)
211 return syntaxCallSt();
212 else{
213 // Специально форсим, будто ждали присваивание
214 eat(IToken::ID);
215 eat(IToken::ASSIGN);
216 }
217 }
218 case IToken::WHILE :
219 return syntaxWhileSt();
220 case IToken::IF :
221 return syntaxIfSt();
222 case IToken::FOR :
223 return syntaxForSt();
224 case IToken::RETURN :
225 return syntaxReturnSt();
226 default:
227 return syntaxEmptySt();
228 }
229}

323std::unique_ptr<IAST> SyntaxAnalyzer::syntaxSelect(void){
324 Token tok = getCurTok();
325 tok.setAdvType(IToken::SELECT);
326 std::unique_ptr<IAST> var = std::move(syntaxVariable());
327 std::vector<std::unique_ptr<IAST>> selections;
328 while(getCurTok().getType() == IToken::LSQBRACKET){
329 eat(IToken::LSQBRACKET);
330 std::unique_ptr<IAST> index = std::move(syntaxExpr());
331 eat(IToken::RSQBRACKET);
332 selections.emplace_back(std::move(index));
333 }
334 if(selections.size() != 1)
335 for(std::size_t i = 0; i < selections.size() - 1; i++){
336 var = ASTFactory::createAST(tok, std::move(var), std::move(selections[i]));
337 }
338 return ASTFactory::createAST(tok, std::move(var), std::move(selections[selections.size()-1]));
339}

384std::unique_ptr<IAST> SyntaxAnalyzer::syntaxFactor(void){
385 /* syntaxFactor ::= ('<INTEGER_CONST>'|'<REAL_CONST>'|'<STRING_CONST>'
386 |'(' syntaxExpr ')'
387 |'+' syntaxFactor| '-' syntaxFactor
388 |'NOT' syntaxFactor
389 |syntaxCallSt| syntaxVariable) */
390 Token token = getCurTok();
391 if(token.getAdvType() == IToken::SOME_CONST){
392 eatAdv(IToken::SOME_CONST);
393 return ASTFactory::createAST(token);
394 }
395 switch(token.getType()) {
396 case IToken::LPAREN :{
397 eat(IToken::LPAREN);
398 std::unique_ptr<IAST> node = syntaxExpr();
399 eat(IToken::RPAREN);
400 return node;
401 }
402
403 case IToken::PLUS :
404 case IToken::MINUS :
405 case IToken::NOT :{
406 eatAdv(IToken::OPERATOR);
407 return ASTFactory::createAST(token, syntaxFactor());
408 }
409
410 case IToken::ID :{
411 if(lookFoward().getType() == IToken::LPAREN)
412 return syntaxCallSt();
413 else if (lookFoward().getType() == IToken::LSQBRACKET)
414 return syntaxSelect();
415 else
416 return syntaxVariable();
417 }
418 default:
419 throw SyntaxException(token, "Во время обработки выражения syntaxFactor встречен неожиданный токен. ");
420 }
421 return nullptr;
422}

424std::unique_ptr<IAST> SyntaxAnalyzer::syntaxIter(void){
425 /* syntaxIter ::= syntaxAssignSt ('to'|'down to') ('<INTEGER_CONST>'|syntaxVariable) */
426 Token varTok = getCurTok();
427 std::unique_ptr<IAST> initSt = syntaxAssignSt();
428
429 Token toTok = getCurTok();
430 if(toTok.getType() == IToken::DOWNTO)
431 eat(IToken::DOWNTO);
432 else
433 eat(IToken::TO);
434
435 Token finTok = getCurTok();
436 if(finTok.getAdvType() != IToken::SOME_CONST and finTok.getType() != IToken::ID)
437 throw AnalyzeException(finTok, "Конечным значением цикла for должна быть целая константа или переменная!");
438 // Пропустим константу или переменную вызовом syntaxFactor
439 syntaxFactor();
440
441 Token condOp = {"<=", IToken::LESS_EQ, IToken::OPERATOR};
442 std::unique_ptr<IAST> varPtr_condition = ASTFactory::createAST(varTok);
443 std::unique_ptr<IAST> finPtr_condition = ASTFactory::createAST(finTok);
444 std::unique_ptr<IAST> condition = ASTFactory::createAST(condOp, std::move(varPtr_condition), std::move(finPtr_condition));
445
446 Token postOp = {"+", IToken::PLUS, IToken::OPERATOR};
447 if(toTok.getType() == IToken::DOWNTO)
448 postOp = {"-", IToken::MINUS, IToken::OPERATOR};
449 Token forOp = {"1", IToken::INTEGER_CONST, IToken::SOME_CONST};
450 std::unique_ptr<IAST> varPtr_postOp = ASTFactory::createAST(varTok);
451 std::unique_ptr<IAST> finPtr_postOp = ASTFactory::createAST(forOp);
452 std::unique_ptr<IAST> postAction = ASTFactory::createAST(postOp, std::move(varPtr_postOp), std::move(finPtr_postOp));
453
454 return ASTFactory::createAST(toTok, std::move(initSt), std::move(condition), std::move(postAction));
455}
