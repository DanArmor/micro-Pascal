73List<Token> Lexer::analyzeProgramText(std::string const &text){

84 Token matched("", IToken::ERROR, IToken::UNKNOWN, lineNum, inLineNum, position);

88 auto wordsEnd = std::sregex_iterator();

91 std::string word = getTrimmed(wordsBegin->str());
92 if(wordsBegin->str().size() > len){
93 len = wordsBegin->str().size();
94 spacesFromBegin = countFromBegin(wordsBegin->str());

99 //std::cout << position << " " << len << " " << offset << " ||| " << text.size() << "\n";
100 if(matched.getType() != IToken::NEWLINE && matched.getStr().size() == 0)
101 throw LexerException(matched, "Неизвестный токен!");
102 position = position + len;
103 inLineNum += len;
104 //std::cout << matched.getInfo() << "\n\n";
105 if(matched.getType() != IToken::NEWLINE && matched.getType() != IToken::COMMENT)
106 tokens.pushBack(matched);
107 if(position >= text.size())
108 break;
109
110 if(matched.getType() == IToken::NEWLINE){
111 lineNum++;
112 inLineNum = 0;

115 lineNum += linesInComment;

117 inLineNum = matched.getStr().size() - matched.getStr().find_last_of('\n') - 1;
118 }
119 }
120 }
121 Token endOfStream("$", IToken::ENDOFSTREAM, IToken::UNKNOWN);
122 tokens.pushBack(endOfStream);
