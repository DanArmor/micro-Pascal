24std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxProgram(void) {
25 /* syntaxProgram ::= 'PROGRAM' 'ID' ';' syntaxBlock '.' */
26 eat(IToken::PROGRAM);
27 Token &nameTok = getCurTok();
28 nameTok.setAdvType(IToken::PROGRAM_NAME);
29
30 eat(IToken::ID);
31 eat(IToken::SEMI);
32 std::vector<std::unique_ptr<IAST>> functions;
33
34 while (isIn(getCurTok().getType(), {IToken::PROCEDURE, IToken::FUNCTION})) {
35 functions.emplace_back(std::move(syntaxFuncDef()));
36 eat(IToken::SEMI);
37 }
38
39 std::unique_ptr<IAST> blockPTR = syntaxBlock();
40 eat(IToken::DOT);
41 return ASTFactory::createAST(nameTok, std::move(functions),
42 std::move(blockPTR));
43}

45std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxFuncDef(void) {
46 Token funcTok = getCurTok();
47 if (funcTok.getType() == IToken::FUNCTION)
48 eat(IToken::FUNCTION);
49 else
50 eat(IToken::PROCEDURE);
51
52 Token &name = getCurTok();
53
54 name.setAdvType(IToken::FUNCTION_NAME);
55 eat(IToken::ID);
56 eat(IToken::LPAREN);
57 std::vector<std::unique_ptr<IAST>> params;
58 while (getCurTok().getType() != IToken::RPAREN) {
59 std::vector<std::unique_ptr<IAST>> sup = std::move(syntaxVarDecl());
60 for (auto &ptr : sup) params.emplace_back(std::move(ptr));
61 if (getCurTok().getType() == IToken::COMMA) eat(IToken::COMMA);
62 }
63 eat(IToken::RPAREN);
64
65 std::unique_ptr<IAST> returnType;
66 if (funcTok.getType() == IToken::FUNCTION) {
67 eat(IToken::COLON);
68 returnType = syntaxTypeSpec();
69 } else {
70 Token typeTok = {"VOID", IToken::VOID, IToken::TYPE_SPEC};
71 returnType = ASTFactory::createAST(typeTok);
72 }
73 eat(IToken::SEMI);
74 std::unique_ptr<IAST> body = std::move(syntaxBlock());
75 return ASTFactory::createAST(name, std::move(params), std::move(returnType),
76 std::move(body));
77}

79std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxBlock(void) {
80 /* syntaxBlock ::= syntaxConsts* syntaxVars* syntaxCompoundSt */
81 std::vector<std::unique_ptr<IAST>> constsList = syntaxConsts();
82 std::vector<std::unique_ptr<IAST>> declsList = syntaxVars();
83 std::unique_ptr<IAST> compoundPTR = syntaxCompoundSt();
84 return ASTFactory::createAST({"", IToken::BLOCK, IToken::UNKNOWN},
85 std::move(constsList), std::move(declsList),
86 std::move(compoundPTR));

89std::vector<std::unique_ptr<IAST>> PascalSyntaxAnalyzer::syntaxConsts(void) {
90 /* syntaxConsts ::= 'CONST' (syntaxConstDecl ';')+ */
91 std::vector<std::unique_ptr<IAST>> constsList;
92 if (getCurTok().getType() == IToken::CONST) {
93 eat(IToken::CONST);
94 while (getCurTok().getType() == IToken::ID) {
95 constsList.emplace_back(syntaxConstDecl());
96 eat(IToken::SEMI);
97 }
98 }
99 return constsList;

102std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxConstDecl(void) {
103 /* syntaxConstDecl ::= '<ID>' '='
104 * ('<INTEGER_CONST>'|'<REAL_CONST>'|'<STRING_CONST>') */
105 Token &tok = getCurTok();
106 tok.setAdvType(IToken::VAR_NAME);
107 std::unique_ptr<IAST> constNamePTR = ASTFactory::createAST(getCurTok());
108 eat(IToken::ID);
109 eat(IToken::EQ);
110
111 Token constValTok = getCurTok();
112 std::unique_ptr<IAST> constValPTR = syntaxExpr();
113
114 constValTok.setAdvType(IToken::AdvType::CONSTDECL);
115
116 return ASTFactory::createAST(constValTok, std::move(constNamePTR),
117 std::move(constValPTR));

120std::vector<std::unique_ptr<IAST>> PascalSyntaxAnalyzer::syntaxVars(void) {
121 /* syntaxVars ::= 'VAR' (syntaxVarDecl ';')+ */
122 std::vector<std::unique_ptr<IAST>> declsList;
123 if (getCurTok().getType() == IToken::VAR) {
124 eat(IToken::VAR);
125 while (getCurTok().getType() == IToken::ID) {
126 std::vector<std::unique_ptr<IAST>> varDeclList = syntaxVarDecl();
127 for (auto &varDeclPTR : varDeclList)
128 declsList.emplace_back(std::move(varDeclPTR));
129 eat(IToken::SEMI);

135std::vector<std::unique_ptr<IAST>> PascalSyntaxAnalyzer::syntaxVarDecl(void) {
136 /* syntaxVarDecl ::= ('<ID>' ','*)+ ':' syntaxTypeSpec */
137 std::vector<std::unique_ptr<IAST>> varsList;
138 Token &tok = getCurTok();
139 tok.setAdvType(IToken::VAR_NAME);
140 varsList.emplace_back(ASTFactory::createAST(tok));
141 eat(IToken::ID);
142
143 while (getCurTok().getType() == IToken::COMMA) {
144 eat(IToken::COMMA);
145 Token &tok = getCurTok();
146 tok.setAdvType(IToken::VAR_NAME);
147 varsList.emplace_back(ASTFactory::createAST(tok));
148 eat(IToken::ID);
149 }
150 eat(IToken::COLON);
151
152 std::unique_ptr<IAST> typeNodePTR(syntaxTypeSpec());
153
154 std::vector<std::unique_ptr<IAST>> varDeclsList;
155 for (auto &var : varsList) {
156 varDeclsList.emplace_back(ASTFactory::createAST(
157 {"", IToken::EMPTY, IToken::VARDECL}, std::move(var),
158 ASTFactory::createCopyOfType(typeNodePTR.get())));
159 }
160 return varDeclsList;
161}

216std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxSt(void) {
217 /* syntaxSt ::=
218 * (syntaxCompoundSt|'<ID>'(syntaxAssignSt|syntaxCallSt)|syntaxWhileSt|syntaxIfSt|syntaxEmptySt)
219 */
220 switch (getCurTok().getType()) {
221 case IToken::BEGIN:
222 return syntaxCompoundSt();
223 case IToken::ID: {
224 if (lookFoward().getType() == IToken::ASSIGN ||
225 lookFoward().getType() == IToken::LSQBRACKET)
226 return syntaxAssignSt();
227 else if (lookFoward().getType() == IToken::LPAREN)
228 return syntaxCallSt();
229 else {
230 // Специально форсим, будто ждали присваивание
231 eat(IToken::ID);
232 eat(IToken::ASSIGN);
233 }
234 }
235 case IToken::WHILE:
236 return syntaxWhileSt();
237 case IToken::IF:
238 return syntaxIfSt();
239 case IToken::FOR:
240 return syntaxForSt();
241 case IToken::RETURN:
242 return syntaxReturnSt();
243 default:
244 return syntaxEmptySt();
245 }
246}

343std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxSelect(void) {
344 Token tok = getCurTok();
345 tok.setAdvType(IToken::SELECT);
346 std::unique_ptr<IAST> var = std::move(syntaxVariable());
347 std::vector<std::unique_ptr<IAST>> selections;
348 while (getCurTok().getType() == IToken::LSQBRACKET) {
349 eat(IToken::LSQBRACKET);
350 std::unique_ptr<IAST> index = std::move(syntaxExpr());
351 eat(IToken::RSQBRACKET);
352 selections.emplace_back(std::move(index));
353 }
354 if (selections.size() != 1)
355 for (std::size_t i = 0; i < selections.size() - 1; i++) {
356 var = ASTFactory::createAST(tok, std::move(var),
357 std::move(selections[i]));
358 }
359 return ASTFactory::createAST(tok, std::move(var),
360 std::move(selections[selections.size() - 1]));
361}

418std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxFactor(void) {
419 /* syntaxFactor ::= ('<INTEGER_CONST>'|'<REAL_CONST>'|'<STRING_CONST>'
420 |'(' syntaxExpr ')'
421 |'+' syntaxFactor| '-' syntaxFactor
422 |'NOT' syntaxFactor
423 |syntaxCallSt| syntaxVariable) */
424 Token token = getCurTok();
425 if (token.getAdvType() == IToken::SOME_CONST) {
426 eatAdv(IToken::SOME_CONST);
427 return ASTFactory::createAST(token);
428 }
429 switch (token.getType()) {
430 case IToken::LPAREN: {
431 eat(IToken::LPAREN);
432 std::unique_ptr<IAST> node = syntaxExpr();
433 eat(IToken::RPAREN);
434 return node;
435 }
436
437 case IToken::PLUS:
438 case IToken::MINUS:
439 case IToken::NOT: {
440 eatAdv(IToken::OPERATOR);
441 return ASTFactory::createAST(token, syntaxFactor());
442 }
443
444 case IToken::ID: {
445 if (lookFoward().getType() == IToken::LPAREN)
446 return syntaxCallSt();
447 else if (lookFoward().getType() == IToken::LSQBRACKET)
448 return syntaxSelect();
449 else
450 return syntaxVariable();
451 }
452 default:
453 throw SyntaxException(token,
454 "Во время обработки выражения syntaxFactor "
455 "встречен неожиданный токен. ");
456 }
457 return nullptr;
458}

460std::unique_ptr<IAST> PascalSyntaxAnalyzer::syntaxIter(void) {
461 /* syntaxIter ::= syntaxAssignSt ('to'|'down to')
462 * ('<INTEGER_CONST>'|syntaxVariable) */
463 Token varTok = getCurTok();
464 std::unique_ptr<IAST> initSt = syntaxAssignSt();
465
466 Token toTok = getCurTok();
467 if (toTok.getType() == IToken::DOWNTO)
468 eat(IToken::DOWNTO);
469 else
470 eat(IToken::TO);
471
472 Token finTok = getCurTok();
473 if (finTok.getAdvType() != IToken::SOME_CONST and
474 finTok.getType() != IToken::ID)
475 throw AnalyzeException(finTok,
476 "Конечным значением цикла for должна быть целая "
477 "константа или переменная!");
478 // Пропустим константу или переменную вызовом syntaxFactor
479 syntaxFactor();
480
481 Token condOp = {"<=", IToken::LESS_EQ, IToken::OPERATOR};
482 std::unique_ptr<IAST> varPtr_condition = ASTFactory::createAST(varTok);
483 std::unique_ptr<IAST> finPtr_condition = ASTFactory::createAST(finTok);
484 std::unique_ptr<IAST> condition = ASTFactory::createAST(
485 condOp, std::move(varPtr_condition), std::move(finPtr_condition));
486
487 Token postOp = {"+", IToken::PLUS, IToken::OPERATOR};
488 if (toTok.getType() == IToken::DOWNTO)
489 postOp = {"-", IToken::MINUS, IToken::OPERATOR};
490 Token forOp = {"1", IToken::INTEGER_CONST, IToken::SOME_CONST};
491 std::unique_ptr<IAST> varPtr_postOp = ASTFactory::createAST(varTok);
492 std::unique_ptr<IAST> finPtr_postOp = ASTFactory::createAST(forOp);
493 std::unique_ptr<IAST> postAction = ASTFactory::createAST(
494 postOp, std::move(varPtr_postOp), std::move(finPtr_postOp));
495
496 return ASTFactory::createAST(toTok, std::move(initSt), std::move(condition),
497 std::move(postAction));
498}
