229 constexpr string_view pretty_name(string_view name) noexcept {
230  for (std::size_t i = name.size(); i > 0; --i) {
231  if (!((name[i - 1] >= '0' && name[i - 1] <= '9') ||
232  (name[i - 1] >= 'a' && name[i - 1] <= 'z') ||
233  (name[i - 1] >= 'A' && name[i - 1] <= 'Z') ||
234 #if defined(MAGIC_ENUM_ENABLE_NONASCII)
235  (name[i - 1] & 0x80) ||
236 #endif
237  (name[i - 1] == '_'))) {
238  name.remove_prefix(i);
239  break;
240  }
241  }
242 
243  if (name.size() > 0 && ((name.front() >= 'a' && name.front() <= 'z') ||
244  (name.front() >= 'A' && name.front() <= 'Z') ||
245 #if defined(MAGIC_ENUM_ENABLE_NONASCII)
246  (name.front() & 0x80) ||
247 #endif
248  (name.front() == '_'))) {
249  return name;
250  }
251 
252  return {}; // Invalid name.

299 template <typename BinaryPredicate>
300 constexpr bool cmp_equal(string_view lhs, string_view rhs, [[maybe_unused]] BinaryPredicate&& p) noexcept(std::is_nothrow_invocable_r_v<bool, BinaryPredicate, char, char>) {
301 #if defined(_MSC_VER) && _MSC_VER < 1920 && !defined(__clang__)
302  // https://developercommunity.visualstudio.com/content/problem/360432/vs20178-regression-c-failed-in-test.html
303  // https://developercommunity.visualstudio.com/content/problem/232218/c-constexpr-string-view.html
304  constexpr bool workaround = true;
305 #else
306  constexpr bool workaround = false;
307 #endif
308  constexpr bool custom_predicate =
309  !std::is_same_v<std::decay_t<BinaryPredicate>, std::equal_to<char>> &&
310  !std::is_same_v<std::decay_t<BinaryPredicate>, std::equal_to<>>;
311 
312  if constexpr (custom_predicate || workaround) {
313  if (lhs.size() != rhs.size()) {
314  return false;
315  }
316 
317  const auto size = lhs.size();
318  for (std::size_t i = 0; i < size; ++i) {
319  if (!p(lhs[i], rhs[i])) {
320  return false;
321  }
322  }
323 
324  return true;
325  } else {
326  return lhs == rhs;
327  }
328 }

330 template <typename L, typename R>
331 constexpr bool cmp_less(L lhs, R rhs) noexcept {
332  static_assert(std::is_integral_v<L> && std::is_integral_v<R>, "magic_enum::detail::cmp_less requires integral type.");
333 
334  if constexpr (std::is_signed_v<L> == std::is_signed_v<R>) {
335  // If same signedness (both signed or both unsigned).
336  return lhs < rhs;
337  } else if constexpr (std::is_same_v<L, bool>) { // bool special case
338  return static_cast<R>(lhs) < rhs;
339  } else if constexpr (std::is_same_v<R, bool>) { // bool special case
340  return lhs < static_cast<L>(rhs);
341  } else if constexpr (std::is_signed_v<R>) {
342  // If 'right' is negative, then result is 'false', otherwise cast & compare.
343  return rhs > 0 && lhs < static_cast<std::make_unsigned_t<R>>(rhs);
344  } else {
345  // If 'left' is negative, then result is 'true', otherwise cast & compare.
346  return lhs < 0 || static_cast<std::make_unsigned_t<L>>(lhs) < rhs;
347  }
348 }
